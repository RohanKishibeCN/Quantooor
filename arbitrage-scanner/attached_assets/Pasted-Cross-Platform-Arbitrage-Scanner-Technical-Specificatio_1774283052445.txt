def detect_cross_chain_arbitrage(self):
    """
    Compare ETH price on Ethereum vs Arbitrum vs Base vs BSC
    Account for: Bridge fees, Bridge time, Reorg risk
    """
    chains = ['ethereum', 'arbitrum', 'base', 'bsc', 'optimism']
    
    for token in ['ETH', 'USDC', 'WBTC']:
        prices = {}
        for chain in chains:
            prices[chain] = get_dex_price(token, chain)
        
        # Find max spread between chains
        for buy_chain in chains:
            for sell_chain in chains:
                if buy_chain == sell_chain: continue
                
                spread = prices[sell_chain] - prices[buy_chain]
                bridge_cost = get_bridge_cost(buy_chain, sell_chain, token)
                
                if spread > bridge_cost * 1.5:  # 50% buffer
                    alert_cross_chain_opp(buy_chain, sell_chain, token)
