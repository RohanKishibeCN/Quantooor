Core Engine (Real-Time Arbitrage Loop)

core/engine.py

import asyncio
import time
from core.filters import validate_opportunity
from core.scorer import score_opportunity
from strategies.cross_exchange import detect_cross_exchange
from strategies.triangular import detect_triangular


class ArbitrageEngine:
    def __init__(self, exchanges, cache):
        self.exchanges = exchanges
        self.cache = cache
        self.last_emit = {}

    async def start(self):
        while True:
            opportunities = []

            # Cross exchange
            opportunities += detect_cross_exchange(self.cache)

            # Triangular
            opportunities += detect_triangular(self.cache)

            final = []
            for op in opportunities:
                if not validate_opportunity(op):
                    continue

                op["score"] = score_opportunity(op)

                key = (op["pair"], op["buy_ex"], op["sell_ex"])

                if key in self.last_emit:
                    if time.time() - self.last_emit[key] < 2:
                        continue

                self.last_emit[key] = time.time()
                final.append(op)

            for op in sorted(final, key=lambda x: x["score"], reverse=True):
                print(op)

            await asyncio.sleep(0.1)
Opportunity Filters (Removes Fake Signals)

core/filters.py

MAX_SPREAD = 6
MIN_SPREAD = 0.2
MAX_RATIO = 8
MIN_LIQ = 8000


def validate_opportunity(op):
    spread = op["spread"]
    ratio = op["sell_price"] / op["buy_price"]

    if spread < MIN_SPREAD:
        return False

    if spread > MAX_SPREAD:
        return False

    if ratio > MAX_RATIO:
        return False

    if op["liquidity"] < MIN_LIQ:
        return False

    return True
Opportunity Scoring Model

Professional scanners rank opportunities, not just detect them.

core/scorer.py

def score_opportunity(op):
    spread_score = op["spread"] * 0.6
    liquidity_score = min(op["liquidity"] / 50000, 1) * 25
    latency_score = max(0, 10 - op["latency_ms"] / 50)

    return spread_score + liquidity_score + latency_score
Orderbook Cache (Ultra Important)

data/orderbook_cache.py

class OrderbookCache:
    def __init__(self):
        self.books = {}

    def update(self, exchange, pair, book):
        self.books.setdefault(pair, {})
        self.books[pair][exchange] = book

    def get_pair(self, pair):
        return self.books.get(pair, {})
Cross-Exchange Arbitrage Detection

strategies/cross_exchange.py

def detect_cross_exchange(cache):
    opportunities = []

    for pair in cache.books:
        books = cache.get_pair(pair)
        exchanges = list(books.keys())

        for buy_ex in exchanges:
            for sell_ex in exchanges:
                if buy_ex == sell_ex:
                    continue

                buy_book = books[buy_ex]
                sell_book = books[sell_ex]

                buy_price = buy_book["asks"][0][0]
                sell_price = sell_book["bids"][0][0]

                if sell_price <= buy_price:
                    continue

                spread = (sell_price - buy_price) / buy_price * 100

                liquidity = min(
                    buy_book["asks"][0][0] * buy_book["asks"][0][1],
                    sell_book["bids"][0][0] * sell_book["bids"][0][1],
                )

                opportunities.append({
                    "type": "cross",
                    "pair": pair,
                    "buy_ex": buy_ex,
                    "sell_ex": sell_ex,
                    "buy_price": buy_price,
                    "sell_price": sell_price,
                    "spread": spread,
                    "liquidity": liquidity,
                    "latency_ms": 20
                })

    return opportunities
Triangular Arbitrage (High-Profit Strategy)

This is where real bots make money.

strategies/triangular.py

def detect_triangular(cache):
    opportunities = []

    for pair in cache.books:
        if "/USDT" not in pair:
            continue

        base = pair.split("/")[0]

        pair1 = f"{base}/USDT"
        pair2 = f"{base}/BTC"
        pair3 = "BTC/USDT"

        if pair1 not in cache.books:
            continue
        if pair2 not in cache.books:
            continue
        if pair3 not in cache.books:
            continue

        books1 = cache.books[pair1]
        books2 = cache.books[pair2]
        books3 = cache.books[pair3]

        for ex in books1:
            if ex not in books2 or ex not in books3:
                continue

            a = books1[ex]["asks"][0][0]
            b = books2[ex]["bids"][0][0]
            c = books3[ex]["bids"][0][0]

            result = (1 / a) * b * c

            if result > 1.002:
                opportunities.append({
                    "type": "triangular",
                    "exchange": ex,
                    "pair": pair,
                    "profit_ratio": result
                })

    return opportunities
WebSocket Exchange Example

This is where speed comes from.

exchanges/kucoin_ws.py

import asyncio
import websockets
import json


class KucoinWS:
    def __init__(self, cache):
        self.cache = cache

    async def connect(self):
        uri = "wss://ws-api.kucoin.com"

        async with websockets.connect(uri) as ws:
            sub = {
                "type": "subscribe",
                "topic": "/market/level2:BTC-USDT",
                "response": True
            }

            await ws.send(json.dumps(sub))

            while True:
                msg = await ws.recv()
                data = json.loads(msg)

                if "data" not in data:
                    continue

                book = {
                    "bids": data["data"]["bids"],
                    "asks": data["data"]["asks"]
                }

                self.cache.update("kucoin", "BTC/USDT", book)
