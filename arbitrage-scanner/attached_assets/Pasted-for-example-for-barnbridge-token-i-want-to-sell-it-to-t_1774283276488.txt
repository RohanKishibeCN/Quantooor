        routes = []
        
        # 1. Check CEX Orderbooks (better for larger amounts due to DEX liquidity constraints)
        for name, exchange in self.exchanges.items():

    async def _analyze_cex(self, exchange, name: str, amount: Decimal) -> Optional[ExitRoute]:
        """Analyze CEX orderbook depth"""
        try:
            symbol = "BOND/USDT"
            orderbook = await exchange.fetch_order_book(symbol)
            
            # Calculate average fill price for the amount
            remaining = amount
            total_cost = Decimal("0")
            liquidity_available = Decimal("0")
            
            for bid in orderbook['bids']:
                price, volume = Decimal(str(bid[0])), Decimal(str(bid[1]))
                liquidity_available += price * volume
                
                if remaining <= 0:
                    break
                    
                fill_qty = min(remaining, volume)
                total_cost += fill_qty * price
                remaining -= fill_qty
            
            if remaining > 0:
                # Not enough liquidity in book
                return None
            
            avg_price = total_cost / amount
            slippage = (avg_price - Decimal(str(orderbook['bids'][0][0]))) / Decimal(str(orderbook['bids'][0][0]))
            
            # CEX fees (taker fee ~0.2%)
            fees = total_cost * Decimal("0.002")
            net = total_cost - fees
            
            return ExitRoute(
                venue=name.upper(),
                price=avg_price,
                liquidity=liquidity_available,
                slippage=slippage,
                fees=fees,
                net_proceeds=net,
                execution_time=30,  # CEX orders fill fast
                recommended=False
            )
            
        except Exception as e:
            print(f"CEX analysis error: {e}")
            return None
